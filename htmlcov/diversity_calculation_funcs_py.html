5def my_data(barcode_amount, n_coefficient, e_coefficient, all_unique): 

6 '''Create array holding the frequency of each barcode in the library. 

7  

8 Keyword arguments:  

9 barcode_amount -- number of unique barcode sequences. 

10 n_coefficient -- parameter for adjusting skew steepness. 

11 e_coefficient -- parameter for adjusting skew strength. 

12 all_unique -- toggle for skew. 

13 ''' 

14 data = np.zeros((barcode_amount, 2)) 

15 data[:,0] = np.arange(1, barcode_amount+1) 

16 data[:,1] = e_coefficient*np.exp(all_unique/n_coefficient*data[:,0])

22def probability_distribution(barcode_distribution): 

23 '''Calculate the probability of picking each barcode.''' 

24 total_barcodes = sum(barcode_distribution[:,1]) 

25 barcode_probability = barcode_distribution[:,1] / total_barcodes 

26 

27 return barcode_probability

29def fraction_unique(barcode_probability, cell_num): 

30 '''Calculate the fraction of uniquely labelled cells. 

31 

32 Keyword arguments:  

33 barcode_probability -- an array of probabilites for picking each barcode. 

34 cell_num -- number of infected cells. 

35 ''' 

36 barcode_contribution = barcode_probability * (1 - (1 - barcode_probability)**(cell_num - 1)) 

37 unique = 1 - sum(barcode_contribution) 

38 

39 return unique
